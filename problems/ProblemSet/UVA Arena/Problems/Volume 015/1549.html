The following C function can be used to calculate the value of D(r) withing a reasonable aumount of time when r is a small integer, say e.g., 1r10, 000. 


long D(long r)
{   long x,y,count=0;
    for(x=-r;x<=r;x++)
        for(y=-r;y<=r;y++)
            if(x*x+y*y<=r*r)
                count++;
    return count;
}
